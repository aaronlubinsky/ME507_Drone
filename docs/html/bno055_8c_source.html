72void BNO_Init(){
73 uint8_t ndof_mode = 0x0C;
74 uint8_t config_mode = 0x00;
75 uint8_t clear_reg = 0x00;
76 uint8_t successfulRead = false;
77 uint8_t sampleData = 0x00;
78 int calibrated = false;
79
80 /* ===== COMMUNICATION VERIFICATION ===== */
81 while(successfulRead == false){
82 // Wait to verify IMU connection
83 HAL_GPIO_TogglePin(GPIOB, GPIO_PIN_14); // Reset BNO055 via hardware pin
84 HAL_Delay(1000); // Wait for BNO055 boot sequence
85 HAL_I2C_DeInit(&hi2c1); // Reset I2C1 peripheral
86 HAL_I2C_Init(&hi2c1); // Reinitialize I2C1
87
88 if (hi2c1.State == HAL_I2C_STATE_READY) {
89 // Once I2C is reset, attempt to read chip ID
90 HAL_I2C_Mem_Read(&hi2c1, BNO055_I2C_ADDR, 0x00, 1, &sampleData, 1, 100);
91 }
92
93 // Verify chip ID (should be 0xA0 for BNO055)
94 if (sampleData == 0xa0){
95 successfulRead = true;
96 }
97 }
98
99 /* ===== MODE CONFIGURATION ===== */
100 // Set to CONFIG mode to allow register writes
101 HAL_I2C_Mem_Write(&hi2c1, BNO055_I2C_ADDR, BNO055_OPR_MODE_ADDR,
102 I2C_MEMADD_SIZE_8BIT, &config_mode, 1, 100);
103 HAL_Delay(25);
104
105 // Set to NDOF mode for full sensor fusion
106 HAL_I2C_Mem_Write(&hi2c1, BNO055_I2C_ADDR, BNO055_OPR_MODE_ADDR,
107 I2C_MEMADD_SIZE_8BIT, &ndof_mode, 1, 100);
108 HAL_Delay(25);
109
110 /* ===== CALIBRATION MONITORING ===== */
111
112 for (int i = 0; i < 18; i++)
113 {
114 HAL_I2C_Mem_Write(&hi2c1,
115 BNO055_I2C_ADDR,
116 BNO055_OFFSET_X + i,
117 I2C_MEMADD_SIZE_8BIT,
118 &clear_reg,
119 1,
120 100);
121 }
122
123 while(calibrated == false){
124 // Wait for IMU to calibrate
125 HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_SET); // Red LED on during calibration
126
127 // Read calibration status register
128 HAL_I2C_Mem_Read(&hi2c1, BNO055_I2C_ADDR, BNO055_CALIB_STAT, 1, &calibData, 1, 100);
129
130 // Check if system calibration is complete (bits 7:6 == 0b11)
131 if (((calibData >> 6) & 0x03) == 0x03){
132 calibrated = true;
133 }
134
135 HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_RESET); // Red LED off
136 }
137}

164void BNO_Read(int32_t *roll, int32_t *pitch, int32_t *yaw){
165 uint8_t eulerData[6];
166 int32_t rawYaw16;
167 int32_t rawPitch16;
168 int32_t rawRoll16;
169
170 /* ===== READ RAW EULER DATA ===== */
171 // Read 6 bytes starting from Euler LSB register
172 HAL_I2C_Mem_Read(&hi2c1, BNO055_I2C_ADDR, BNO055_EULER_LSB,
173 I2C_MEMADD_SIZE_8BIT, eulerData, 6, 100);
174
175 /* ===== DATA CONVERSION ===== */
176 // Combine LSB and MSB bytes to form 16-bit signed values
177 rawYaw16 = (int16_t)((eulerData[1] << 8) | eulerData[0]); // Bytes 0-1: Yaw
178 rawRoll16 = (int16_t)((eulerData[3] << 8) | eulerData[2]); // Bytes 2-3: Roll
179 rawPitch16 = (int16_t)((eulerData[5] << 8) | eulerData[4]); // Bytes 4-5: Pitch
180
181 // Convert from 1/16 degree resolution to millidegrees
182 *yaw = ((int32_t)rawYaw16 * 1000) / 16;
183 *roll = (((int32_t)rawRoll16 * 1000) / 16) - roll_offset;
184 *pitch = (((int32_t)rawPitch16 * 1000) / 16) - pitch_offset;
185
186 /* ===== FLIGHT DATA LOGGING ===== */
187 // Log data to blackbox at specified frequency
188 if (counter++ == blackboxFreq) {
189 if (sample_index < MAX_SAMPLES) {
190 blackbox[sample_index].pitch = *pitch;
191 blackbox[sample_index].roll = *roll;
192 blackbox[sample_index].pitchSet = pitch_set;
193 blackbox[sample_index].rollSet = roll_set;
194 sample_index++;
195 }
196 counter = 0;
197 }
198}
