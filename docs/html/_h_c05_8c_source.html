102 int32_t LjoyX, LjoyY, RjoyX, LT, RT, ENTER;

104 /* ===== INPUT VALIDATION ===== */
105 // Skip pound sign if present
106 if (charBuf[0] == '#') {
107 charBuf++; // Move pointer to the next character
108 } else {
109 badBTcount++;
110 printf("Bad BT input (did not process): %d\r\n", badBTcount);
111 return;
112 }

114 /* ===== DATA PARSING ===== */
115 // Parse comma-separated values using strtok
116 char *token = strtok(charBuf, ",");
117 if (token) LjoyX = (int32_t)strtol(token, NULL, 10);

132 if (token) ENTER = (int32_t)strtol(token, NULL, 10);

134 /* ===== CONTROL MAPPING ===== */
135 // Convert joystick values to flight control commands
136
137 // Roll/Pitch: Scale joystick input to ±20° (assuming joystick range ±1000)
138 *roll = LjoyX * 180 / 9; // Scale to millidegrees (±20000)
139 *pitch = LjoyY * 180 / 9; // Scale to millidegrees (±20000)
140
141 // Yaw: Relative control - add rate command to current heading
142 *yaw = yaw_true + (RjoyX) / 10;
143
144 // Handle yaw wraparound (0-360°)
145 if (*yaw < 0) {
146 *yaw = *yaw + 360000;
147 } else if (*yaw > 360000) {
148 *yaw = *yaw - 360000;
149 }
150
151 // Throttle: Differential trigger control (RT increases, LT decreases)
152 *effort = *effort + (RT - LT) * effortRate / 1000;
153
154 // Constrain effort to safe limits
155 stopFlag = false;
156 if (*effort <= 0) {
157 *effort = 0;
158 stopFlag = true;
159 } else if (*effort > 1000) {
160 *effort = 1000;
161 }

163 // Blackbox dump trigger
164 if (ENTER == 1) {

166 }
