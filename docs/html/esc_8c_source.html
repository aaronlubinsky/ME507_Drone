130void armESC()
131{
132 effort_set = 1000;
133
134 while(roll_set < 10000){
135 // Start PWM generation on all timer channels
136 HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_1);
137 HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_2);
138 HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_3);
139 HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_4);
140
141 // Calculate arming PWM value (approximately 1000μs pulse width)
142 armCompare = effort_set*4 - 2000;
143
144 // Clamp arming value to safe range
145 if (armCompare < 960) armCompare = 960;
146 if (armCompare > 2000) armCompare = 2000;
147
148 // Set PWM compare values for all motors
149 __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, armCompare);
150 __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_2, armCompare);
151 __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_3, armCompare);
152 __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_4, armCompare);
153
154 // Toggle status LED during arming
155 HAL_GPIO_TogglePin(GPIOA, GPIO_PIN_0);
156 HAL_Delay(125);
157 }
158
159 // Reset effort and compare values after arming
160 effort_set = 0;
161 armCompare = 0;
162}

188void update_Motors()
189{
190 // PWM Mapping: Compare 960 = 1ms (0%), Compare 2000 = 2ms (100%)
191
192 /* ===== ROLL PID CALCULATION ===== */
193 roll_error = -roll_set + roll_true;
194
195 roll_integral += roll_error/1000; //Wind up protection for integral
196 if (roll_integral > max_integral) {
197 roll_integral = max_integral;
198 } else if (roll_integral < -max_integral) {
199 roll_integral = -max_integral;
200 }
201
202
203 // Shift array left and add new delta at end
204 float roll_delta = roll_error - last_roll_error;
205 for (int i = 0; i < DERIVATIVE_SAMPLES - 1; i++) {
206 roll_delta_history[i] = roll_delta_history[i + 1];
207 }
208 roll_delta_history[DERIVATIVE_SAMPLES - 1] = roll_delta;
209
210 // Calculate rolling average derivative
211 float roll_derivative_sum = 0;
212 for (int i = 0; i < DERIVATIVE_SAMPLES; i++) {
213 roll_derivative_sum += roll_delta_history[i];
214 }
215 roll_derivative = roll_derivative_sum / DERIVATIVE_SAMPLES;
216
217 // TESTING ONLY - Roll effort disabled
218 roll_effort = -(Kp_roll * roll_error + Ki_roll * roll_integral + Kd_roll * roll_derivative) / PID_SCALE;
219 last_roll_error = roll_error;
220
221 /* ===== PITCH PID CALCULATION ===== */
222 pitch_error = -pitch_set + pitch_true;
223 pitch_integral += pitch_error/1000; //Wind up protection for integral
224 if (pitch_integral > max_integral) {
225 pitch_integral = max_integral;
226 } else if (pitch_integral < -max_integral) {
227 pitch_integral = -max_integral;
228 }
229
230 // Shift array left and add new delta at end
231 float pitch_delta = pitch_error - last_pitch_error;
232 for (int i = 0; i < DERIVATIVE_SAMPLES - 1; i++) {
233 pitch_delta_history[i] = pitch_delta_history[i + 1];
234 }
235 pitch_delta_history[DERIVATIVE_SAMPLES - 1] = pitch_delta;
236
237 // Calculate rolling average derivative
238 float pitch_derivative_sum = 0;
239 for (int i = 0; i < DERIVATIVE_SAMPLES; i++) {
240 pitch_derivative_sum += pitch_delta_history[i];
241 }
242 pitch_derivative = pitch_derivative_sum / DERIVATIVE_SAMPLES;
243
244 pitch_effort = -(Kp_pitch * pitch_error + Ki_pitch * pitch_integral + Kd_pitch * pitch_derivative) / PID_SCALE;
245 last_pitch_error = pitch_error;
246
247 /* ===== YAW PID CALCULATION ===== */
248 yaw_error = -yaw_set + yaw_true;
249 yaw_integral += yaw_error;
250
251 // Shift array left and add new delta at end
252 float yaw_delta = yaw_error - last_yaw_error;
253 for (int i = 0; i < DERIVATIVE_SAMPLES - 1; i++) {
254 yaw_delta_history[i] = yaw_delta_history[i + 1];
255 }
256 yaw_delta_history[DERIVATIVE_SAMPLES - 1] = yaw_delta;
257
258 // Calculate rolling average derivative
259 float yaw_derivative_sum = 0;
260 for (int i = 0; i < DERIVATIVE_SAMPLES; i++) {
261 yaw_derivative_sum += yaw_delta_history[i];
262 }
263 yaw_derivative = yaw_derivative_sum / DERIVATIVE_SAMPLES;
264
265 yaw_effort = -(Kp_yaw * yaw_error + Ki_yaw * yaw_integral + Kd_yaw * yaw_derivative) / PID_SCALE;
266 last_yaw_error = yaw_error;
267
268 /* ===== BASE THROTTLE CALCULATION ===== */
269 // Start with base throttle plus individual motor offsets
270 A = effort_set * K_effort/PID_SCALE + motA_offset;
271 B = effort_set * K_effort/PID_SCALE + motB_offset;
272 C = effort_set * K_effort/PID_SCALE + motC_offset;
273 D = effort_set * K_effort/PID_SCALE + motD_offset;
274
275 /* ===== CONTROL MIXING ===== */
276 // Apply pitch control (affects front/rear motor pairs)
277 if (pitch_effort > 0) {
278 A += pitch_effort; // Front motors get more power
279 D += pitch_effort;
280 }
281 else if (pitch_effort < 0) {
282 B -= pitch_effort; // Rear motors get more power
283 C -= pitch_effort;
284 }
285
286 // Apply roll control (affects left/right motor pairs)
287 if (roll_effort > 0) {
288 A += roll_effort; // Left motors get more power
289 B += roll_effort;
290 }
291 else if (roll_effort < 0) {
292 C -= roll_effort; // Right motors get more power
293 D -= roll_effort;
294 }
295
296 // Apply yaw control (affects diagonal motor pairs)
297 if (yaw_effort > 0) {
298 B += yaw_effort; // CW motors get more power
299 D += yaw_effort;
300 }
301 else if (yaw_effort < 0) {
302 A -= yaw_effort; // CCW motors get more power
303 C -= yaw_effort;
304 }
305
306
307 /* ===== SAFETY LIMITS ===== */
308 // Clamp all motors between 960 (0%) and 1500 (≈50% for safety)
309 if ((A < 960) | (stopFlag == true)) A = 960;
310 if (A > 1600) A = 1700;
311
312 if ((B < 960) | (stopFlag == true)) B = 960;
313 if (B > 1700) B = 1700;
314
315 if ((C < 960) | (stopFlag == true)) C = 960;
316 if (C > 1700) C = 1700;
317
318 if ((D < 960) | (stopFlag == true)) D = 960;
319 if (D > 1700) D = 1700;
320
321 // Debug output (commented)

325 // Update timer compare registers to set motor speeds

327 __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_2, B);
